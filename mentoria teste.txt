<script type="module">
    import { db } from './firebase-config.js';
    import { ref, onValue } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-database.js";

    // 1. Carregar Resumo de Produção
    onValue(ref(db, 'estudos_completos'), (snapshot) => {
        const dados = snapshot.val();
        if (dados) {
            const examesCompletos = Object.keys(dados);
            document.getElementById('total-pecas').innerText = examesCompletos.length;

            const progressoDiv = document.getElementById('progresso-exames');
            progressoDiv.innerHTML = '';
           
            examesCompletos.forEach(ex => {
                progressoDiv.innerHTML += `
                    <div class="progresso-item">
                        <span>Exame ${ex}</span>
                        <span class="status-badge status-completo">PRODUZIDO</span>
                    </div>
                `;
            });
        }
    });

    // 2. Carregar Post-its (Dicas)
    onValue(ref(db, 'dicas_postits'), (snapshot) => {
        const dicas = snapshot.val();
        const lista = document.getElementById('lista-postits');
        if (dicas) {
            lista.innerHTML = '';
            Object.values(dicas).forEach(d => {
                lista.innerHTML += `<div class="postit-item"><i class="fas fa-lightbulb"></i> ${d.texto} <br><small style="opacity:0.7">${d.peca || ''}</small></div>`;
            });
        }
    });
</script>
